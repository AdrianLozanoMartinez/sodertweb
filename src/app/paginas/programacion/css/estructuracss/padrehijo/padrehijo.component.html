Html

{{co}}!DOCTYPE html>
    {{co}}html lang="en">
      {{co}}head> 
        {{co}}meta charset="UTF-8">
         {{co}}meta name="viewport" content="width=device-width, initial-scale=1.0">
         {{co}}title>Título de pestaña{{co}}/title>
         {{co}}link rel="stylesheet" href="ArchivoCssAparte.css">
      {{co}}/head>
      {{co}}body>
          {{co}}div>{{co}}p>Párrafo Padre{{co}}/p>{{co}}/div>      {{co}}!--Si el parrafo esta dentro del div. Solo a sus hijos directos-->
          {{co}}p>Párrafo Hijo{{co}}/p>              {{co}}!-- Despues del div el párrafo continuo (mismo padre), los siguientes no -->
          {{co}}p>Párrafos restantes{{co}}/p>        {{co}}!-- Todos los párrafos restantes --> 
      {{co}}/body>
    {{co}}/html>

Css

/*ORDEN PARA QUE FUNCIONE*/
div>p{{c}}                  /*Si el parrafo esta dentro del div. Solo a sus hijos directos - Párrafo Padre*/
    color:yellow;
}
div~p{{c}}                  /*Todos los párrafos restantes - Párrafos restantes*/
    color: green;
}
div+p{{c}}                  /*Despues del div el párrafo continuo (mismo padre), los siguientes no - Párrafo Hijo*/
    color:red;
}

Html

{{co}}!DOCTYPE html>
{{co}}html lang="en">
  {{co}}head> 
    {{co}}meta charset="UTF-8">
     {{co}}meta name="viewport" content="width=device-width, initial-scale=1.0">
     {{co}}title>Título de pestaña{{co}}/title>
     {{co}}link rel="stylesheet" href="ArchivoCssAparte.css">
  {{co}}/head>
  {{co}}body>
      {{co}}div>
      {{co}}div id="x1">Hover (Poner ratón encima y ver párrafo oculto)      {{co}}!--Hover-->
          {{co}}p>El Div oculta este párrafo con hover{{co}}/p>
      {{co}}/div> 

      {{co}}div id="x2">{{co}}/div>                                               {{co}}!--Elemento vacío, empty-->

      {{co}}div id="x4">                                                     {{co}}!--1º y último hijo, efectos-->
          {{co}}p>first-child (Primer hijo){{co}}/p>
          {{co}}p>Hijo del medio{{co}}/p>
          {{co}}p>last-child (Último hijo){{co}}/p>
      {{co}}/div>

      {{co}}/div>

      {{co}}p id="par">Párrafo fuera del div{{co}}/p>                               
      {{co}}p class="der">Odd: Párrafo Impar{{co}}/p>    
      {{co}}p class="doc">Even: Párrafo Par{{co}}/p>                              {{co}}!--(::) Para especificar el elemento que se encuentra dentro, en este caso el párrafo--> 
  {{co}}/body>
{{co}}/html>

Css

#x1{{c}}
    width: 500px;
    height: 100px;
}
#x1>p{{c}}                           /*Primer p*/
    display:none;                /*oculto*/
    background-color: yellow;
}
#x1:hover p{{c}}                     /*Pasa el raton y sale el párrafo (p)*/
    display:block;
}
#x4 p:last-child{{c}}                /*Último hijo, último párrafo*/
    background-color: aqua;
}
#x4 p:first-child{{c}}               /*Primer hijo, primer párrafo*/
    background-color: blueviolet;
}
div:empty{{c}}                       /*Elemento vacío en el html, el x2*/
    height: 200px;
    width: 200px;
    background-color: red;
    margin-top: 5px;
}
#par{{c}}                             /*Justifica el párrafo y lo encierra en las medidas de width y height*/
    width: 200px;
    height: 50px;
    text-align: justify;
}
#par::first-line{{c}}                /*(::) Para especificar el elemento que se encuentra dentro, en este caso el párrafo*/
    text-transform: uppercase;
}

.der:nth-child(odd){{c}}                  /*Hijos. Impar*/
