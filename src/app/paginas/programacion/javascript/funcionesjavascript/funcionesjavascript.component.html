Con parámetros


UN parámetro
const algo = datos => {{c}}           //Opcional llevar ()
  console.log( datos );
}
algo('Ponemos un dato');



2 o más parámetro
const algo = ( dato1, dato2 ) => {{c}}           //Siempre debe llevar ()
console.log( dato1 + dato2 );
}
algo("Ponemos un dato: ", 1985);

Sin recibir datos

const algo = () => {{c}}
    return 'Texto para sacar'
}

const resultado = algo();
console.log( resultado );


Simplificamos

const algo = () => "Texto para sacar";

const resultado = algo();
console.log( resultado );

Recibiendo un dato

const algo = dato => {{c}}
  return 'Texto de la ' + dato;
};

const resultado = algo( 'función' );
console.log( resultado );


Simplificamos

const algo = dato => "Texto para " + dato;

const resultado = algo("sacar");
console.log( resultado );

Recibiendo datos

const algo = ( dato1, dato2 ) => {{c}}
  return dato1 + dato2;
};

const resultado = algo( "Ponemos un dato: ", 1985 );
console.log( resultado );


Simplificamos

const algo = (dato1, dato2) => (dato1 + dato2);

const resultado = algo("Ponemos un dato: ", 1985);
console.log( resultado );
